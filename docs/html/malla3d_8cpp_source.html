295bool Malla3D::RayIntersectsTriangle(glm::vec3 rayOrigin,
296 glm::vec3 rayVector,
297 glm::vec3 vertex0,
298 glm::vec3 vertex1,
299 glm::vec3 vertex2,
300 glm::vec3& outIntersectionPoint){
301 const float EPSILON = 0.00000001;
302 glm::vec3 edge1, edge2, h, s, q;
303 float a,f,u,v;
304 edge1 = vertex1 - vertex0;
305 edge2 = vertex2 - vertex0;
306 h = glm::cross(rayVector,edge2);
307 a = glm::dot(edge1,h);
308
309 if (a > -EPSILON && a < EPSILON){
310 return false;
311 }
312
313 f = 1.0/a;
314 s = rayOrigin - vertex0;
315 u = f * glm::dot(s,h);
316
317 if (u < 0.0 || u > 1.0){
318 return false;
319 }
320
321 q = glm::cross(s,edge1);
322 v = f * glm::dot(rayVector,q);
323
324 if (v < 0.0 || u + v > 1.0){
325 return false;
326 }
327
328 float t = f * glm::dot(edge2,q);
329
330 if (t > EPSILON){
331 outIntersectionPoint = rayOrigin + rayVector * t;
332 return true;
333 }
334 else{
335 return false;
336 }

863 float m = 0.4 * height_sym;
864 float n = 0.1 * width_sym;
865
866 float sym = 0;
867 float sym_diff = 0;
868 int index_up, index_down;
869 float up, down;
870
871 std::vector<float> sumas;
872
873 for(int w = 0; w < width_sym; w++){
874 sym = 0;
875 for(int h = (height_sym/2 - m); h <= (height_sym/2 + m); h++){
876 sym_diff = 0;
877 for(int x = 1; x <= n; x++){
878 index_up = w+x;
879 index_down = w-x;
880 if(index_down < 0){
881 index_down = (int)(width_sym + index_down);
882 }
883 if(index_up > (int)width_sym){
884 index_up = index_up % (int)width_sym;
885 }
886
887 up = panorama[h][index_up] / 255.0;
888 down = panorama[h][index_down] / 255.0;
889 sym_diff += abs(down - up);
890 }
891 sym_diff = sym_diff / n;
892 sym += sym_diff;
893 }
894
895 sym = sym / (2*m);
896 sym = 1.0 - sym;
897 sumas.push_back(sym);
898 }
899
900 return sumas;
